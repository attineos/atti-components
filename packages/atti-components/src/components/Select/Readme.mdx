Select :

```js
<Select id="single" options={['< 1 year', 'Between 1 and 3 years', '> 3 years']} />
```

Select with alphabetical sort :

```js
const Example = () => {
  const [value, setValue] = useState('')

  return (
    <div>
      <div> Selected value sorted : {value}</div>
      <Select
        sort="ASC"
        id="single"
        onChange={e => setValue(e.target.value)}
        options={['allo', 'coco', 'bobo']}
      />
    </div>
  )
}

render(<Example />)
```

Select multiple :

```js
const Example = () => {
  const [values, setValues] = useState([])

  const onChangeMultipleValue = () => {
    const options = document.getElementById('multiple')
    setValues(
      reduce(
        options,
        (acc, { selected, value }) => {
          if (selected) {
            acc.push(value)
          }
          return acc
        },
        [],
      ).join(', '),
    )
  }

  return (
    <div>
      <div> Selected values : {values} (press ctrl to select multiple values)</div>
      <Select
        id="multiple"
        onChange={onChangeMultipleValue}
        options={['value 1', 'value 2', 'value 3', 'value4', 'value5']}
        isMulti
      />
    </div>
  )
}

render(<Example />)
```
