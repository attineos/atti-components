Displays a stepper that can be filled.

## Examples

```js
const Example = () => {
  const [step, setStep] = useState(1)
  const handleStep = () => {
    setStep(2)
    var elmt = document.getElementById(step)
    elmt.style.backgroundColor = 'darkblue'
  }

  return (
    <div style={{ paddingBottom: '80px' }}>
      <Stepper>
        <Stepper.Step id={1} onClick={handleStep}>
          <Stepper.Element />
          <Stepper.Label>1</Stepper.Label>
          <Stepper.Label>Num 1</Stepper.Label>
        </Stepper.Step>
        <Stepper.Step id={2} onClick={handleStep} style={{ marginLeft: '400px' }}>
          <Stepper.Element />
          <Stepper.Label>2</Stepper.Label>
          <Stepper.Label>Num 2</Stepper.Label>
        </Stepper.Step>
      </Stepper>
    </div>
  )
}

render(<Example />)
```
