```js
  <Stepper>
    <Stepper.Step id="1">
      <Stepper.Element />
      <Stepper.Label style={{ color: 'white' }}>1</Stepper.Label>
      <Stepper.Label>Num 1</Stepper.Label>
    </Stepper.Step>
    <Stepper.Step id="2">
      <Stepper.Element />
      <Stepper.Label style={{ color: 'white' }}>2</Stepper.Label>
      <Stepper.Label>Num 2</Stepper.Label>
    </Stepper.Step>
    <Stepper.Step id="3">
      <Stepper.Element />
      <Stepper.Label style={{ color: 'white' }}>3</Stepper.Label>
      <Stepper.Label>Num 3</Stepper.Label>
    </Stepper.Step>
  </Stepper>
</div>
```
