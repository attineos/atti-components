Displays a stepper that can be adapted.

## Examples

```js
<Stepper>
  <Stepper.Step id="1">
    <Stepper.Label middle>1</Stepper.Label>
    <Stepper.Label>Num 1</Stepper.Label>
  </Stepper.Step>
  <Stepper.Step id="2">
    <Stepper.Label middle>2</Stepper.Label>
    <Stepper.Label>Num 2</Stepper.Label>
  </Stepper.Step>
  <Stepper.Step id="3">
    <Stepper.Label middle>3</Stepper.Label>
    <Stepper.Label>Num 3</Stepper.Label>
  </Stepper.Step>
</Stepper>
```
