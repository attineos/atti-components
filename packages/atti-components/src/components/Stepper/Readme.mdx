Displays a stepper that can be filled.

## Examples

```js
const Example = () => {
  const [step, setStep] = useState(1)
  const handleStep = () => {
    var elmt1 = document.getElementById(step)
    var elmt2 = document.getElementById(step - 1)
    var elmt3 = document.getElementById(step + 1)
    if (step == 1) {
      elmt1.style.backgroundColor = '#0E7C86'
      elmt3.style.backgroundColor = '#2CB1BC'
      setStep(step + 1)
    } else {
      elmt2.style.backgroundColor = '#2CB1BC'
      elmt1.style.backgroundColor = '#0E7C86'
      setStep(step - 1)
    }
  }

  return (
    <div style={{ padding: '30px', display: 'flex', alignItems: 'center' }}>
      <Stepper>
        <Stepper.Step onClick={handleStep}>
          <Stepper.Element id={1} />
          <Stepper.Label style={{ color: 'white' }}>1</Stepper.Label>
          <Stepper.Label>Num 1</Stepper.Label>
        </Stepper.Step>
        <Stepper.Step onClick={handleStep}>
          <Stepper.Element id={2} />
          <Stepper.Label style={{ color: 'white' }}>2</Stepper.Label>
          <Stepper.Label>Num 2</Stepper.Label>
        </Stepper.Step>
      </Stepper>
    </div>
  )
}

render(<Example />)
```
