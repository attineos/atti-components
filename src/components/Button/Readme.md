Button :

```js
<Button onClick={() => console.log('Hello !')}>Normal</Button>
```

Animations can be added when clicking on the button :

```js
<ButtonGroup>
  <Button>default</Button>
  <Button animation="circleGrow">circleGrow</Button>
  <Button animation="flash">flash</Button>
  <Button animation="focus">focus</Button>
  <Button animation="shine">shine</Button>
  <Button animation="slideBottom">slideBottom</Button>
  <Button animation="slideLeft">slideLeft</Button>
  <Button animation="slideOpenDiag">slideOpenDiag</Button>
  <Button animation="slideOpenHorizontal">slideOpenHorizontal</Button>
  <Button animation="slideOpenVertical">slideOpenVertical</Button>
  <Button animation="slideRight">slideRight</Button>
  <Button animation="slideTop">slideTop</Button>
</ButtonGroup>
```
